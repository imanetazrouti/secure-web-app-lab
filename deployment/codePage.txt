<?php
$host = "localhost";
$db   = "test_db";
$user = "php_user";
$pass = "php_password";

$message = "";

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $fullname = $_POST['fullname'] ?? '';
    $username = $_POST['username'] ?? '';
    $password = $_POST['password'] ?? '';
    $email    = $_POST['email'] ?? '';
    $phone    = $_POST['phone'] ?? '';

    // Connect to MariaDB
    $conn = new mysqli($host, $user, $pass, $db);
    if ($conn->connect_error) die("Connection failed: " . $conn->connect_error);

    // Insert user
    $stmt = $conn->prepare("INSERT INTO users (fullname, username, password, email, phone) VALUES (?, ?, ?, ?, ?)");
    $stmt->bind_param("sssss", $fullname, $username, $password, $email, $phone);

    if ($stmt->execute()) $message = "Registration successful! 🎉";
    else $message = "Error: " . $stmt->error;

    $stmt->close();
    $conn->close();
}
